attribute vec3 v_position;
attribute vec2 v_uv;

varying vec3 color;
uniform mat4 mvpMatrix;
uniform mat3 m_inverse;

void main(void){
	vec3 normalDirection = normalize( , gl_Normal );
	vec3 lightDirection = normalize( glm::vec3(5.0, 5.0, 5.0) );

	vec3 diffuseReflection = vec3( 1.0,1.0,1.0 ) * vec3(v_uv, 1.0) * max(0.0, dot(normalDirection, lightDirection) );

	gl_Position = mvpMatrix * vec4(v_position, 1.0);
	color = v_color;
}
